import tkinter as tk from tkinter import ttk, messagebox import re class DataFilterEngine: """محرك معالجة وترشيح البيانات - Backend Logic""" @staticmethod def clean_input(raw_data): # تقسيم البيانات بناءً على الفواصل وتنظيف الفراغات if not raw_data.strip(): return [] return [item.strip() for item in re.split(r'[,\n]+', raw_data) if item.strip()] @staticmethod def filter_data(items, filter_type, criterion=""): result = [] if filter_type == "الأعداد الزوجية (Even Numbers)": result = [x for x in items if x.isdigit() and int(x) % 2 == 0] elif filter_type == "الأعداد الفردية (Odd Numbers)": result = [x for x in items if x.isdigit() and int(x) % 2 != 0] elif filter_type == "أكبر من (Greater Than)": if not criterion.isdigit(): raise ValueError("يجب إدخال عدد صحيح في خانة الشرط.") result = [x for x in items if x.isdigit() and int(x) > int(criterion)] elif filter_type == "تبدأ بحرف (Starts With)": if not criterion: raise ValueError("الرجاء تحديد حرف الترشيح.") result = [x for x in items if x.lower().startswith(criterion.lower())] elif filter_type == "طول الكلمة أكبر من (Length >)": if not criterion.isdigit(): raise ValueError("يجب إدخال عدد صحيح لطول الكلمة.") result = [x for x in items if len(x) > int(criterion)] return result class ModernFilterApp(tk.Tk): """الواجهة الرسومية الحديثة للمشروع - UI View""" def __init__(self): super().__init__() self.title("منظومة ترشيح البيانات المتقدمة v2.0") self.geometry("600x520") self.configure(bg="#1e1e2e") # ثيم داكن احترافي self.setup_styles() self.create_widgets() def setup_styles(self): self.style = ttk.Style() self.style.theme_use('clam') # تخصيص مظهر القوائم والمدخلات لتبدو حديثة self.style.configure("TCombobox", fieldbackground="#252538", background="#252538", foreground="white") self.style.configure("TEntry", fieldbackground="#252538", foreground="white") def create_widgets(self): # العنوان الرئيسي title_lbl = tk.Label(self, text="Advanced Data Filtering System", font=("Segoe UI", 16, "bold"), bg="#1e1e2e", fg="#cba6f7") title_lbl.pack(pady=15) # منطقة الإدخال input_frame = tk.LabelFrame(self, text=" المدخلات (افصل بين العناصر بفاصلة) ", bg="#1e1e2e", fg="#bac2de", font=("Segoe UI", 10)) input_frame.pack(pady=10, padx=20, fill=tk.X) self.txt_input = tk.Text(input_frame, height=4, bg="#252538", fg="#cdd6f4", insertbackground="white", font=("Segoe UI", 11), bd=0, padx=5, pady=5) self.txt_input.pack(fill=tk.X, padx=10, pady=10) self.txt_input.insert("1.0", "10, apple, 23, orange, 8, banana, 105, python, 42") # منطقة التحكم والشروط control_frame = tk.Frame(self, bg="#1e1e2e") control_frame.pack(pady=10, padx=20, fill=tk.X) tk.Label(control_frame, text="نوع الترشيح:", bg="#1e1e2e", fg="#bac2de").grid(row=0, column=0, sticky=tk.W, padx=5) self.filter_options = [ "الأعداد الزوجية (Even Numbers)", "الأعداد الفردية (Odd Numbers)", "أكبر من (Greater Than)", "تبدأ بحرف (Starts With)", "طول الكلمة أكبر من (Length >)" ] self.combo_filter = ttk.Combobox(control_frame, values=self.filter_options, state="readonly", width=28) self.combo_filter.current(0) self.combo_filter.grid(row=1, column=0, padx=5, pady=5) tk.Label(control_frame, text="المعيار / الشرط:", bg="#1e1e2e", fg="#bac2de").grid(row=0, column=1, sticky=tk.W, padx=5) self.entry_criterion = ttk.Entry(control_frame, width=15) self.entry_criterion.grid(row=1, column=1, padx=5, pady=5) # زر التشغيل الاحترافي self.btn_run = tk.Button(self, text="تطبيق نظام الترشيح ⚡", font=("Segoe UI", 11, "bold"), bg="#a6e3a1", fg="#11111b", activebackground="#94e2d5", bd=0, cursor="hand2", command=self.process_filtering) self.btn_run.pack(pady=15, ipadx=20, ipady=5) # منطقة المخرجات output_frame = tk.LabelFrame(self, text=" البيانات المصفّاة والنتائج ", bg="#1e1e2e", fg="#a6e3a1", font=("Segoe UI", 10, "bold")) output_frame.pack(pady=10, padx=20, fill=tk.BOTH, expand=True) self.txt_output = tk.Text(output_frame, height=4, bg="#161622", fg="#a6e3a1", font=("Segoe UI", 12, "bold"), state=tk.DISABLED, bd=0, padx=5, pady=5) self.txt_output.pack(fill=tk.BOTH, expand=True, padx=10, pady=10) def process_filtering(self): raw_data = self.txt_input.get("1.0", tk.END) items = DataFilterEngine.clean_input(raw_data) if not items: messagebox.showwarning("تنبيه", "صندوق المدخلات فارغ!") return filter_type = self.combo_filter.get() criterion = self.entry_criterion.get().strip() try: # استدعاء محرك الفلترة filtered_results = DataFilterEngine.filter_data(items, filter_type, criterion) # تحديث الواجهة بالنتائج self.txt_output.config(state=tk.NORMAL) self.txt_output.delete("1.0", tk.END) if filtered_results: self.txt_output.insert(tk.END, " | ".join(filtered_results)) else: self.txt_output.insert(tk.END, "❌ لم يتم العثور على بيانات تطابق هذا الشرط.") self.txt_output.config(state=tk.DISABLED) except ValueError as e: messagebox.showerror("خطأ في المعطيات", str(e)) if __name__ == "__main__": app = ModernFilterApp() app.mainloop()
